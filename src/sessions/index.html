const checkout = await AdyenCheckout({
    environment: 'test',
    clientKey: 'test_...',

        console.info(result, component);
    },
    onError: (error, component) => {
        console.error(error.name, error.message, error.stack, component);
    }
});
                            
const dropin = checkout
    .create('dropin')
